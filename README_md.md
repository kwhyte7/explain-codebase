# Function to sum two numbers

def add(a, b):
    """
    Add two numbers and return the result.

    Parameters:
    a (int/float): First number
    b (int/float): Second number

    Returns:
    int/float: Sum of the two numbers
    """
    return a + b

# Example usage
result = add(5, 3)
print(result)  # Output: 8
